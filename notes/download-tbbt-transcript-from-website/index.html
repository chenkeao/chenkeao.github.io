import requests
from bs4 import BeautifulSoup
import os

url_list = []

def download_script(url):
  response = requests.get(url)
  response.encoding = "utf-8"
  soup = BeautifulSoup(response.text, "html.parser")

  script_text = ""
  title = soup.find("h2", class_="title").get_text()
  script_text = script_text + "## " + title + "\n\n"
  div = soup.find("div", class_="entrytext")
  p_tags = div.find_all("p")

  for p in p_tags:
	  text = p.get_text()
	  print(text)
	  try:
		  before, after = text.split(":", 1)
		  script_text = script_text + "**" + before + "**" + ": " + after + "\n\n"
	  except:
		  script_text = script_text + text + "\n\n"

  return script_text


with open("TBBT.md", "a", encoding="utf-8") as f:
  for url in url_list:
	  print(download_script(url), file=f)
